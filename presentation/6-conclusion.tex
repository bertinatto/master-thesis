\begin{frame}
    \frametitle{Limitations}
    \begin{itemize}
        \item eBPF limitations
            \begin{itemize}
                \item Requires elevated privileges
                \item Verifier strictness
                \item Complex real-time analysis is hard in kernel space
                \item Deployment of solution is complex
            \end{itemize}
        \item Research scope
            \begin{itemize}
                \item Narrow range of security scenarios covered
                \item eBPF-based programs introduce an additional layer, which can present extra security risks
            \end{itemize}
    \end{itemize}
\end{frame}
